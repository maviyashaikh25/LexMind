Indian Kanoon API scraper

pythonsrc/scraper.py
import requests, json, time, os
from dotenv import load_dotenv

load_dotenv()
API_KEY = os.getenv("INDIAN_KANOON_API_KEY")
BASE_URL = "https://api.indiankanoon.org"

def search_cases(query: str, pages: int = 50) -> list:
    """Search and return case documents from Indian Kanoon."""
    headers = {"Authorization": f"Token {API_KEY}"}
    all_docs = []

    for page in range(pages):
        resp = requests.post(
            f"{BASE_URL}/search/",
            data={"formInput": query, "pagenum": page},
            headers=headers, timeout=10
        )
        if resp.status_code != 200:
            print(f"Error page {page}: {resp.status_code}")
            break
        data = resp.json()
        docs = data.get("docs", [])
        if not docs:
            break
        all_docs.extend(docs)
        time.sleep(0.5)   # respect rate limit

    return all_docs

def fetch_full_doc(doc_id: str) -> dict:
    """Fetch full text of a single judgment."""
    headers = {"Authorization": f"Token {API_KEY}"}
    resp = requests.post(
        f"{BASE_URL}/doc/{doc_id}/",
        headers=headers, timeout=15
    )
    return resp.json() if resp.status_code == 200 else {}

if __name__ == "__main__":
    queries = [
        "criminal appeal supreme court section 302",
        "civil breach of contract high court",
        "cheating fraud section 420 IPC",
        "dowry death section 304B IPC",
        "bail application criminal case",
    ]

    os.makedirs("data/raw", exist_ok=True)
    for q in queries:
        print(f"Scraping: {q}")
        docs = search_cases(q, pages=20)
        full_docs = []
        for doc in docs[:10]:  # limit to 10 docs per query for demo
            doc_id = doc.get("tid")
            if doc_id:
                full_text_data = fetch_full_doc(str(doc_id))
                if full_text_data:
                    doc["doc"] = full_text_data.get("doc", "")
                    full_docs.append(doc)
                time.sleep(0.5)
        fname = q.replace(" ", "_")[:40]
        with open(f"data/raw/{fname}.json", "w") as f:
            json.dump(full_docs, f, indent=2)
        print(f"  Saved {len(full_docs)} docs with text")

Data preprocessing pipeline

pythonsrc/preprocess.py
import re, json, pandas as pd
from pathlib import Path
from langchain.text_splitter import RecursiveCharacterTextSplitter

# ── Text cleaning ──────────────────────────────────────
def clean_text(text: str) -> str:
    text = re.sub(r'&\w+;', ' ', text)          # HTML entities
    text = re.sub(r'<[^>]+>', ' ', text)        # HTML tags
    text = re.sub(r'\n{3,}', '\n\n', text)       # excess newlines
    text = re.sub(r'[ \t\r\f\v]+', ' ', text)    # horizontal whitespace
    text = re.sub(r'(?i)IN THE.{0,60}COURT', '', text)  # boilerplate
    return text.strip()

# ── Outcome label extraction ───────────────────────────
OUTCOME_MAP = {
    "appeal allowed": 1,
    "petition allowed": 1,
    "appeal dismissed": 0,
    "petition dismissed": 0,
    "partly allowed": 2,
    "disposed of": 2,
}

def extract_outcome(text: str) -> int:
    text_lower = text.lower()[-500:]           # check last 500 chars
    for phrase, label in OUTCOME_MAP.items():
        if phrase in text_lower:
            return label
    return -1                                   # -1 = unknown

# ── Metadata extraction ────────────────────────────────
def extract_metadata(doc: dict) -> dict:
    publish_date = doc.get("publishdate") or "2000"
    return {
        "doc_id": doc.get("tid", "") or "",
        "title": doc.get("title", "") or "",
        "court": doc.get("docsource", "") or "",
        "date": doc.get("publishdate", "") or "",
        "citation": doc.get("citation", "") or "",
        "text_length": len(doc.get("doc") or ""),
        "year": str(publish_date)[:4],
    }

# ── Chunker for RAG ────────────────────────────────────
splitter = RecursiveCharacterTextSplitter(
    chunk_size=512,
    chunk_overlap=50,
    separators=["\n\n", "\n", ". ", " "]
)

def chunk_document(doc_id: str, text: str, metadata: dict) -> list:
    chunks = splitter.split_text(text)
    chunk_meta = {k: v for k, v in metadata.items() if k != "cleaned_text"}
    return [
        {"chunk_id": f"{doc_id}_c{i}",
         "text": c,
         "doc_id": doc_id,
         **chunk_meta}
        for i, c in enumerate(chunks)
    ]

# ── Main pipeline ──────────────────────────────────────
if __name__ == "__main__":
    all_chunks, metadata_rows = [], []

    for fpath in Path("data/raw").glob("*.json"):
        with open(fpath) as f:
            docs = json.load(f)

        for doc in docs:
            raw_text = doc.get("doc") or ""
            if len(raw_text) < 200: continue

            clean = clean_text(raw_text)
            meta  = extract_metadata(doc)
            meta["outcome"] = extract_outcome(clean)
            meta["cleaned_text"] = clean

            chunks = chunk_document(meta["doc_id"], clean, meta)
            all_chunks.extend(chunks)
            metadata_rows.append(meta)

    pd.DataFrame(metadata_rows).to_csv("data/cases_metadata.csv", index=False)
    with open("data/processed/chunks.jsonl", "w") as f:
        for c in all_chunks:
            f.write(json.dumps(c) + "\n")

    print(f"Done: {len(metadata_rows)} cases → {len(all_chunks)} chunks")

NLP module

Train spaCy NER — legal entity extractor

What this module does Extracts structured entities from free-form case text: who are the parties, which statute was invoked, who is the judge, what is the verdict. These entities feed both the ML feature builder and the UI entity panel.
pythonsrc/train_ner.py
import spacy, json, random
from spacy.training import Example
from spacy.util import minibatch, compounding
from pathlib import Path

# ── Load pre-annotated data (from iiitnlp/Indian-Legal-NER) ──
ID_TO_LABEL = {
    0: "O",
    1: "B-PETITIONER", 2: "I-PETITIONER",
    3: "B-RESPONDENT", 4: "I-RESPONDENT",
    5: "B-JUDGE", 6: "I-JUDGE",
    7: "B-COURT", 8: "I-COURT",
    9: "B-STATUTE", 10: "I-STATUTE",
    11: "B-SECTION", 12: "I-SECTION",
    13: "B-CITATION", 14: "I-CITATION",
    15: "B-VERDICT", 16: "I-VERDICT",
    17: "B-DATE", 18: "I-DATE",
    19: "B-CHARGE", 20: "I-CHARGE"
}

def load_ner_data(path: str) -> list:
    """Convert HuggingFace NER dataset to spaCy format."""
    data = []
    with open(path) as f:
        for line in f:
            row = json.loads(line)
            tokens = row["tokens"]
            raw_tags = row["ner_tags"]
            tags = [ID_TO_LABEL.get(t, "O") if isinstance(t, int) else t for t in raw_tags]
            text   = " ".join(tokens)
            entities = []
            start, i = 0, 0
            while i < len(tokens):
                token_start = text.find(tokens[i], start)
                if token_start == -1:
                    token_start = start
                token_end   = token_start + len(tokens[i])
                if tags[i] != "O":
                    label = tags[i].split("-")[-1]  # B-JUDGE → JUDGE
                    entities.append((token_start, token_end, label))
                start = token_end + 1
                i += 1
            data.append((text, {"entities": entities}))
    return data

# ── Add custom labels ─────────────────────────────────────────
CUSTOM_LABELS = [
    "PETITIONER", "RESPONDENT", "JUDGE", "COURT",
    "STATUTE", "SECTION", "CITATION", "VERDICT",
    "DATE", "CHARGE"
]

def train_ner(train_path: str, output_dir: str, n_iter: int = 30):
    nlp = spacy.load("en_core_web_sm")
    ner = nlp.get_pipe("ner")

    for label in CUSTOM_LABELS:
        ner.add_label(label)

    train_data = load_ner_data(train_path)
    random.shuffle(train_data)

    optimizer = nlp.resume_training()
    other_pipes = [p for p in nlp.pipe_names if p != "ner"]

    with nlp.disable_pipes(*other_pipes):
        for epoch in range(n_iter):
            random.shuffle(train_data)
            losses = {}
            batches = minibatch(train_data, size=compounding(4.0, 32.0, 1.001))
            for batch in batches:
                for text, annotations in batch:
                    doc = nlp.make_doc(text)
                    example = Example.from_dict(doc, annotations)
                    nlp.update([example], sgd=optimizer, losses=losses, drop=0.3)
            print(f"Epoch {epoch+1}/{n_iter} — NER loss: {losses.get('ner', 0):.4f}")

    nlp.to_disk(output_dir)
    print(f"Model saved to {output_dir}")

if __name__ == "__main__":
    train_ner(
        train_path="data/raw/ner_train.jsonl",
        output_dir="models/legal_ner",
        n_iter=30
    )

# ── Inference usage ───────────────────────────────────────────
def extract_entities(text: str, nlp=None) -> dict:
    if nlp is None:
        nlp = spacy.load("models/legal_ner")
    doc = nlp(text)
    result = {}
    for ent in doc.ents:
        result.setdefault(ent.label_, []).append(ent.text)
    return result

Fine-tune BART for legal summarization

pythonsrc/train_summarizer.py
from transformers import (BartForConditionalGeneration,
                          BartTokenizer, Seq2SeqTrainer,
                          Seq2SeqTrainingArguments, DataCollatorForSeq2Seq)
from datasets import load_dataset
import evaluate, numpy as np

MODEL_NAME = "facebook/bart-large-cnn"
tokenizer  = BartTokenizer.from_pretrained(MODEL_NAME)
model      = BartForConditionalGeneration.from_pretrained(MODEL_NAME)

# ── Load LegalSum dataset ─────────────────────────────────────
dataset = load_dataset("d0r1h/LegalSum")   # judgment + headnote pairs
if "test" not in dataset:
    dataset = dataset["train"].train_test_split(test_size=0.1)

def preprocess(batch):
    inputs = tokenizer(
        batch["judgment"], max_length=1024,
        truncation=True, padding="max_length"
    )
    labels = tokenizer(
        batch["summary"], max_length=256,
        truncation=True, padding="max_length"
    )
    inputs["labels"] = [
        [(-100 if t == tokenizer.pad_token_id else t) for t in ids]
        for ids in labels["input_ids"]
    ]
    return inputs

tokenized = dataset.map(preprocess, batched=True, remove_columns=dataset["train"].column_names)

# ── ROUGE metric ──────────────────────────────────────────────
rouge = evaluate.load("rouge")

def compute_metrics(eval_pred):
    preds, labels = eval_pred
    decoded_preds   = tokenizer.batch_decode(preds, skip_special_tokens=True)
    labels[labels == -100] = tokenizer.pad_token_id
    decoded_labels  = tokenizer.batch_decode(labels, skip_special_tokens=True)
    result = rouge.compute(predictions=decoded_preds, references=decoded_labels, use_stemmer=True)
    return {k: round(v * 100, 4) for k, v in result.items()}

# ── Training config ───────────────────────────────────────────
args = Seq2SeqTrainingArguments(
    output_dir="models/legal_summarizer",
    num_train_epochs=5,
    per_device_train_batch_size=4,
    per_device_eval_batch_size=4,
    warmup_steps=500,
    weight_decay=0.01,
    learning_rate=2e-5,
    predict_with_generate=True,
    evaluation_strategy="epoch",
    save_strategy="epoch",
    load_best_model_at_end=True,
    fp16=True,                        # use GPU half-precision
    logging_steps=100,
)

collator = DataCollatorForSeq2Seq(tokenizer, model=model, padding=True)
trainer = Seq2SeqTrainer(
    model=model,
    args=args,
    train_dataset=tokenized["train"],
    eval_dataset=tokenized["test"],
    tokenizer=tokenizer,
    data_collator=collator,
    compute_metrics=compute_metrics,
)

trainer.train()
model.save_pretrained("models/legal_summarizer")
tokenizer.save_pretrained("models/legal_summarizer")
print("Summarizer saved.")

# ── Inference ─────────────────────────────────────────────────
def summarize(text: str, model=None, tokenizer=None, max_len: int = 200) -> str:
    if model is None or tokenizer is None:
        from transformers import BartForConditionalGeneration, BartTokenizer
        tokenizer = BartTokenizer.from_pretrained("models/legal_summarizer")
        model = BartForConditionalGeneration.from_pretrained("models/legal_summarizer")
    inputs = tokenizer(text, return_tensors="pt", max_length=1024, truncation=True)
    ids    = model.generate(inputs["input_ids"], max_length=max_len, num_beams=4, early_stopping=True)
    return tokenizer.decode(ids[0], skip_special_tokens=True)

XGBoost outcome predictor + SHAP explainability

pythonsrc/train_outcome.py
import pandas as pd, numpy as np
import xgboost as xgb, shap, json
from sklearn.model_selection import StratifiedKFold, cross_val_score
from sklearn.preprocessing import LabelEncoder
from sklearn.metrics import classification_report

# ── Feature engineering ───────────────────────────────────────
COURT_RANK = {
    "Supreme Court": 4, "High Court": 3,
    "District Court": 2, "Tribunal": 1
}
CHARGE_LABELS = ["302", "307", "420", "498A", "376", "304B", "other"]

def build_features(df: pd.DataFrame) -> pd.DataFrame:
    df["court_level"]   = df["court"].map(COURT_RANK).fillna(1)
    df["year_int"]      = pd.to_numeric(df["year"], errors="coerce").fillna(2000)
    df["text_len"]      = df["text_length"].fillna(0)
    df["has_bail"]      = df["title"].str.contains("bail", case=False).astype(int)
    df["has_appeal"]    = df["title"].str.contains("appeal", case=False).astype(int)
    df["is_criminal"]   = df["title"].str.contains("criminal|IPC|CrPC", case=False).astype(int)

    # One-hot encode top charge sections
    for ch in CHARGE_LABELS:
        df[f"charge_{ch}"] = df.get("citation", "").str.contains(ch).astype(int)

    FEATURE_COLS = [
        "court_level", "year_int", "text_len",
        "has_bail", "has_appeal", "is_criminal"
    ] + [f"charge_{ch}" for ch in CHARGE_LABELS]

    return df[FEATURE_COLS], FEATURE_COLS

# ── Load data ─────────────────────────────────────────────────
df = pd.read_csv("data/cases_metadata.csv")
df = df[df["outcome"] >= 0].copy()       # remove unlabeled

X, feat_cols = build_features(df)
y = df["outcome"].values

# ── Train XGBoost ─────────────────────────────────────────────
model = xgb.XGBClassifier(
    n_estimators=500,
    learning_rate=0.05,
    max_depth=6,
    subsample=0.8,
    colsample_bytree=0.8,
    use_label_encoder=False,
    eval_metric="mlogloss",
    random_state=42
)

skf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
cv_scores = cross_val_score(model, X, y, cv=skf, scoring="f1_macro")
print(f"CV F1-macro: {cv_scores.mean():.3f} ± {cv_scores.std():.3f}")

model.fit(X, y)
model.save_model("models/outcome_xgb.json")
json.dump(feat_cols, open("models/feature_cols.json", "w"))

# ── Feature engineering for a single query ────────────────────
def build_features_single(entities_dict: dict, court: str, feat_names: list) -> dict:
    court_level = COURT_RANK.get(court, 1)
    title_text = " ".join(entities_dict.get("VERDICT", []) + entities_dict.get("CHARGE", []))
    citation_text = " ".join(entities_dict.get("CITATION", []))
    
    features = {
        "court_level": court_level,
        "year_int": 2024,
        "text_len": 1000,
        "has_bail": int("bail" in title_text.lower()),
        "has_appeal": int("appeal" in title_text.lower()),
        "is_criminal": int(any(x in title_text.lower() for x in ["criminal", "ipc", "crpc"]))
    }
    
    for ch in CHARGE_LABELS:
        features[f"charge_{ch}"] = int(ch in citation_text)
        
    return features

# ── Prediction + explanation for one case ─────────────────────
LABELS = {0: "Appeal Dismissed", 1: "Appeal Allowed", 2: "Partly Allowed"}

def predict_outcome(features_dict: dict, xgb_model=None, feat_names=None) -> dict:
    if xgb_model is None:
        xgb_model = xgb.XGBClassifier()
        xgb_model.load_model("models/outcome_xgb.json")
    if feat_names is None:
        feat_names = json.load(open("models/feature_cols.json"))
        
    x_row = np.array([[features_dict.get(f, 0) for f in feat_names]])
    proba = xgb_model.predict_proba(x_row)[0]
    pred  = int(np.argmax(proba))
    
    sv = shap.TreeExplainer(xgb_model).shap_values(x_row)
    if isinstance(sv, list):
        sv_pred = sv[pred][0]
    elif len(sv.shape) == 3:
        sv_pred = sv[0, :, pred]
    else:
        sv_pred = sv[0]
        
    top_feats = sorted(
        zip(feat_names, sv_pred), key=lambda x: abs(x[1]), reverse=True
    )[:5]
    
    return {
        "label": LABELS[pred],
        "confidence": round(float(proba[pred]), 3),
        "probabilities": {LABELS[i]: round(float(p), 3) for i, p in enumerate(proba)},
        "top_factors": [{"feature": f, "shap": round(float(s), 4)} for f, s in top_feats],
    }

Step 6a — Embed and index the corpus

pythonsrc/embed_corpus.py
import json, os
from pinecone import Pinecone, ServerlessSpec
from sentence_transformers import SentenceTransformer
from tqdm import tqdm
from dotenv import load_dotenv

load_dotenv()
pc = Pinecone(api_key=os.getenv("PINECONE_API_KEY"))

# Use InLegalBERT — fine-tuned on Indian legal text
embedder = SentenceTransformer("law-ai/InLegalBERT")
INDEX_NAME = "lexmind-cases"

# Create index if it doesn't exist
if INDEX_NAME not in [i.name for i in pc.list_indexes()]:
    pc.create_index(
        name=INDEX_NAME,
        dimension=768,          # InLegalBERT output size
        metric="cosine",
        spec=ServerlessSpec(cloud="aws", region="us-east-1")
    )

index = pc.Index(INDEX_NAME)

# ── Batch embed and upsert ────────────────────────────────────
def upsert_chunks(chunks_path: str, batch_size: int = 100):
    chunks = []
    with open(chunks_path) as f:
        chunks = [json.loads(l) for l in f]

    for i in tqdm(range(0, len(chunks), batch_size), desc="Upserting"):
        batch = chunks[i:i+batch_size]
        texts   = [c["text"] for c in batch]
        vectors = embedder.encode(texts, show_progress_bar=False)

        upsert_data = [
            (
                c["chunk_id"],
                vectors[j].tolist(),
                {
                    "doc_id": c["doc_id"],
                    "court": c.get("court", ""),
                    "year": c.get("year", ""),
                    "citation": c.get("citation", ""),
                    "text": c["text"][:500],    # store preview
                }
            )
            for j, c in enumerate(batch)
        ]
        index.upsert(vectors=upsert_data)

    print(f"Indexed {len(chunks)} chunks into Pinecone.")

if __name__ == "__main__":
    upsert_chunks("data/processed/chunks.jsonl")

Step 6b — Retrieval chain with LangChain

pythonsrc/rag_chain.py
import os
from pinecone import Pinecone
from langchain_anthropic import ChatAnthropic
from langchain.prompts import PromptTemplate
from langchain.chains import RetrievalQAWithSourcesChain
from langchain_community.vectorstores import Pinecone as PineconeVS
from langchain_huggingface import HuggingFaceEmbeddings
from dotenv import load_dotenv

load_dotenv()

# ── Connect to existing Pinecone index ────────────────────────
embeddings = HuggingFaceEmbeddings(model_name="law-ai/InLegalBERT")
vectorstore = PineconeVS.from_existing_index("lexmind-cases", embeddings)

# ── Retriever with metadata filtering ─────────────────────────
def get_retriever(court_filter: str = None, year_from: int = 2000):
    search_kwargs = {"k": 20}
    if court_filter:
        search_kwargs["filter"] = {"court": {"$eq": court_filter}}
    return vectorstore.as_retriever(
        search_type="mmr",          # max marginal relevance — diversity
        search_kwargs=search_kwargs
    )

# ── Legal RAG prompt ──────────────────────────────────────────
PROMPT = PromptTemplate(
    input_variables=["summaries", "question"],
    template="""You are an expert Indian legal research assistant.

Use ONLY the retrieved case excerpts below to answer the question.
For every claim, cite the case name and year in parentheses.
If you are unsure, say "Based on available precedents, this is unclear."
Be concise — 3 to 5 sentences maximum.

Retrieved cases:
{summaries}

Question: {question}

Answer (with citations):"""
)

# ── Build the chain ───────────────────────────────────────────
llm = ChatAnthropic(
    model="claude-sonnet-4-20250514",
    api_key=os.getenv("ANTHROPIC_API_KEY"),
    max_tokens=1000
)

def get_rag_chain(court_filter=None):
    retriever = get_retriever(court_filter)
    return RetrievalQAWithSourcesChain.from_chain_type(
        llm=llm,
        retriever=retriever,
        chain_type="stuff",
        chain_type_kwargs={"prompt": PROMPT},
        return_source_documents=True
    )

def run_rag(query: str, court: str = None) -> dict:
    chain  = get_rag_chain(court)
    result = chain.invoke({"question": query})
    sources = [
        {
            "citation": doc.metadata.get("citation", "Unknown"),
            "court": doc.metadata.get("court", ""),
            "year": doc.metadata.get("year", ""),
            "preview": doc.page_content[:300]
        }
        for doc in result["source_documents"][:5]  # top 5 only
    ]
    return {"answer": result["answer"], "sources": sources}

import spacy, xgboost as xgb, json
from transformers import BartForConditionalGeneration, BartTokenizer

from src.rag_chain import run_rag
from src.train_outcome import predict_outcome, build_features_single
from src.train_ner import extract_entities
from src.train_summarizer import summarize

# ── Startup: load all models once ─────────────────────────────
models = {}

@asynccontextmanager
async def lifespan(app: FastAPI):
    print("Loading models...")
    models["ner"] = spacy.load("models/legal_ner")
    models["xgb"] = xgb.XGBClassifier()
    models["xgb"].load_model("models/outcome_xgb.json")
    models["bart_tok"] = BartTokenizer.from_pretrained("models/legal_summarizer")
    models["bart"]     = BartForConditionalGeneration.from_pretrained("models/legal_summarizer")
    models["feat_cols"] = json.load(open("models/feature_cols.json"))
    print("All models loaded.")
    yield

app = FastAPI(title="LexMind API", version="1.0.0", lifespan=lifespan)

# ── Request / Response schemas ────────────────────────────────
class CaseQuery(BaseModel):
    case_brief: str
    jurisdiction: str = "Supreme Court"
    year_from: int = 2000

class LexMindResponse(BaseModel):
    summary: str
    entities: dict
    precedents: list
    outcome: dict
    answer: str

# ── Health check ──────────────────────────────────────────────
@app.get("/")
async def health():
    return {"status": "ok", "models_loaded": list(models.keys())}

# ── Main analyze endpoint ─────────────────────────────────────
@app.post("/analyze", response_model=LexMindResponse)
async def analyze_case(query: CaseQuery):
    try:
        # 1. NLP: entity extraction
        entities = extract_entities(query.case_brief, models["ner"])

        # 2. DL: summarize brief
        summary = summarize(
            query.case_brief,
            models["bart"], models["bart_tok"]
        )

        # 3. ML: outcome prediction with SHAP
        features = build_features_single(
            entities, query.jurisdiction, models["feat_cols"]
        )
        outcome = predict_outcome(features, models["xgb"], models["feat_cols"])

        # 4. RAG: retrieve precedents + LLM answer
        rag_result = run_rag(query.case_brief, court=query.jurisdiction)

        return LexMindResponse(
            summary=summary,
            entities=entities,
            precedents=rag_result["sources"],
            outcome=outcome,
            answer=rag_result["answer"]
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
